## Part 1: gRPC Bridge Implementation

### Node-to-Node Communication

JOTP processes communicate across nodes using gRPC bridges:

```java
package io.github.seanchatmangpt.jotp.examples.distributed;

import io.github.seanchatmangpt.jotp.*;
import io.grpc.*;
import io.grpc.stub.StreamObserver;
import java.time.Duration;
import java.util.concurrent.ConcurrentHashMap;

/**
 * GrpcProcBridge - Enables JOTP processes to communicate across nodes
 * using gRPC for transport while maintaining process semantics.
 */
public class GrpcProcBridge {

    // ── Protocol Buffers Definition ───────────────────────────────────────────

    // proto/proc_bridge.proto
    /*
    syntax = "proto3";

    package jotp;

    service ProcBridge {
      rpc Tell(TellRequest) returns (TellResponse);
      rpc Ask(AskRequest) returns (AskResponse);
      rpc Stream(StreamRequest) returns (stream StreamMessage);
    }

    message TellRequest {
      string node_id = 1;
      string process_id = 2;
      bytes message = 3;
    }

    message TellResponse {
      bool success = 1;
      string error = 2;
    }

    message AskRequest {
      string node_id = 1;
      string process_id = 2;
      bytes message = 3;
      int64 timeout_ms = 4;
    }

    message AskResponse {
      bool success = 1;
      bytes response = 2;
      string error = 3;
    }

    message StreamRequest {
      string node_id = 1;
      string process_id = 2;
    }

    message StreamMessage {
      bytes message = 1;
    }
    */

    // ── Remote Process Reference ──────────────────────────────────────────────

    public static class RemoteProcRef<S, M> {
        private final String nodeId;
        private final String processId;
        private final ProcBridgeGrpc.ProcBridgeBlockingStub stub;

        public RemoteProcRef(String nodeId, String processId, Channel channel) {
            this.nodeId = nodeId;
            this.processId = processId;
            this.stub = ProcBridgeGrpc.newBlockingStub(channel);
        }

        public void tell(M message) {
            var request = ProcProto.TellRequest.newBuilder()
                .setNodeId(nodeId)
                .setProcessId(processId)
                .setMessage(serialize(message))
                .build();

            var response = stub.tell(request);

            if (!response.getSuccess()) {
                throw new RuntimeException("Remote tell failed: " + response.getError());
            }
        }

        public S ask(M message, Duration timeout) {
            var request = ProcProto.AskRequest.newBuilder()
                .setNodeId(nodeId)
                .setProcessId(processId)
                .setMessage(serialize(message))
                .setTimeoutMs(timeout.toMillis())
                .build();

            var response = stub.ask(request);

            if (!response.getSuccess()) {
                throw new RuntimeException("Remote ask failed: " + response.getError());
            }

            return deserialize(response.getResponse());
        }

        private byte[] serialize(M message) {
            // Implementation: Serialize to bytes (JSON, protobuf, etc.)
            return new byte[0];
        }

        private S deserialize(byte[] data) {
            // Implementation: Deserialize from bytes
            return null;
        }
    }

    // ── gRPC Service Implementation ───────────────────────────────────────────

    public static class ProcBridgeService extends ProcBridgeGrpc.ProcBridgeImplBase {
        private final String nodeId;
        private final ProcRegistry registry;

        public ProcBridgeService(String nodeId, ProcRegistry registry) {
            this.nodeId = nodeId;
            this.registry = registry;
        }

        @Override
        public void tell(ProcProto.TellRequest request, StreamObserver<ProcProto.TellResponse> responseObserver) {
            try {
                var procRef = registry.get(request.getProcessId());

                if (procRef.isEmpty()) {
                    responseObserver.onNext(ProcProto.TellResponse.newBuilder()
                        .setSuccess(false)
                        .setError("Process not found: " + request.getProcessId())
                        .build());
                    responseObserver.onCompleted();
                    return;
                }

                var message = deserialize(request.getMessage());
                procRef.get().tell(message);

                responseObserver.onNext(ProcProto.TellResponse.newBuilder()
                    .setSuccess(true)
                    .build());
                responseObserver.onCompleted();

            } catch (Exception e) {
                responseObserver.onNext(ProcProto.TellResponse.newBuilder()
                    .setSuccess(false)
                    .setError(e.getMessage())
                    .build());
                responseObserver.onCompleted();
            }
        }

        @Override
        public void ask(ProcProto.AskRequest request, StreamObserver<ProcProto.AskResponse> responseObserver) {
            try {
                var procRef = registry.get(request.getProcessId());

                if (procRef.isEmpty()) {
                    responseObserver.onNext(ProcProto.AskResponse.newBuilder()
                        .setSuccess(false)
                        .setError("Process not found: " + request.getProcessId())
                        .build());
                    responseObserver.onCompleted();
                    return;
                }

                var message = deserialize(request.getMessage());
                var timeout = Duration.ofMillis(request.getTimeoutMs());
                var response = procRef.get().ask(message, timeout).get();

                responseObserver.onNext(ProcProto.AskResponse.newBuilder()
                    .setSuccess(true)
                    .setResponse(serialize(response))
                    .build());
                responseObserver.onCompleted();

            } catch (Exception e) {
                responseObserver.onNext(ProcProto.AskResponse.newBuilder()
                    .setSuccess(false)
                    .setError(e.getMessage())
                    .build());
                responseObserver.onCompleted();
            }
        }

        private Object deserialize(ByteString bytes) {
            // Implementation
            return null;
        }

        private ByteString serialize(Object obj) {
            // Implementation
            return ByteString.EMPTY;
        }
    }

    // ── Node Registry ─────────────────────────────────────────────────────────

    public static class NodeRegistry {
        private final String localNodeId;
        private final ConcurrentHashMap<String, Channel> nodeChannels = new ConcurrentHashMap<>();
        private final ConcurrentHashMap<String, RemoteProcRef<?, ?>> remoteRefs = new ConcurrentHashMap<>();

        public NodeRegistry(String localNodeId) {
            this.localNodeId = localNodeId;
        }

        public void registerNode(String nodeId, String host, int port) {
            var channel = ManagedChannelBuilder.forAddress(host, port)
                .usePlaintext()
                .build();

            nodeChannels.put(nodeId, channel);
            System.out.println("[NodeRegistry] Registered node: " + nodeId);
        }

        public <S, M> RemoteProcRef<S, M> getRemoteRef(String nodeId, String processId) {
            var key = nodeId + ":" + processId;

            @SuppressWarnings("unchecked")
            var cached = (RemoteProcRef<S, M>) remoteRefs.get(key);

            if (cached != null) {
                return cached;
            }

            var channel = nodeChannels.get(nodeId);
            if (channel == null) {
                throw new IllegalArgumentException("Unknown node: " + nodeId);
            }

            var ref = new RemoteProcRef<S, M>(nodeId, processId, channel);
            remoteRefs.put(key, ref);
            return ref;
        }

        public void shutdown() {
            nodeChannels.values().forEach(Channel::shutdown);
        }
    }

    // ── Example Usage ────────────────────────────────────────────────────────

    public static void main(String[] args) throws Exception {
        // Node 1: US-East
        var node1Registry = ProcRegistry.create();
        var node1Bridge = new ProcBridgeService("node-1", node1Registry);

        var node1 = ServerBuilder.forPort(50051)
            .addService(node1Bridge)
            .build()
            .start();

        // Node 2: US-West
        var node2Registry = ProcRegistry.create();
        var node2Bridge = new ProcBridgeService("node-2", node2Registry);

        var node2 = ServerBuilder.forPort(50052)
            .addService(node2Bridge)
            .build()
            .start();

        // Register nodes
        var registry1 = new NodeRegistry("node-1");
        registry1.registerNode("node-2", "localhost", 50052);

        var registry2 = new NodeRegistry("node-2");
        registry2.registerNode("node-1", "localhost", 50051);

        // Create processes
        sealed interface Msg {}
        record Request(String data) implements Msg {}
        record Response(String result) implements Msg {}

        var proc1 = Proc.spawn(0, (state, msg) -> state);
        node1Registry.register("proc-1", proc1);

        var proc2 = Proc.spawn(0, (state, msg) -> state);
        node2Registry.register("proc-2", proc2);

        // Cross-node communication
        var remoteProc = registry1.getRemoteRef("node-2", "proc-2");
        remoteProc.tell(new Request("Hello from node-1!"));

        Thread.sleep(1000);

        node1.shutdown();
        node2.shutdown();
        registry1.shutdown();
        registry2.shutdown();
    }
}
```

## Part 2: Distributed Supervision Trees

### Pattern: Geo-Replicated Supervisor

```java
package io.github.seanchatmangpt.jotp.examples.distributed;

import io.github.seanchatmangpt.jotp.*;
import java.time.Duration;
import java.util.List;

/**
 * DistributedSupervisor - Multi-region supervision tree with
 * automatic failover and consistency guarantees.
 */
public class DistributedSupervisor {

    // ── Message Types ────────────────────────────────────────────────────────

    sealed interface DistributedMsg permits
        ReplicateState,
        StateUpdate,
        NodeJoin,
        NodeLeave,
        ElectLeader,
        SyncMetadata {}

    record ReplicateState(String key, Object value) implements DistributedMsg {}
    record StateUpdate(String key, Object value, long version) implements DistributedMsg {}
    record NodeJoin(String nodeId, String host, int port) implements DistributedMsg {}
    record NodeLeave(String nodeId) implements DistributedMsg {}
    record ElectLeader() implements DistributedMsg {}
    record SyncMetadata(List<String> activeNodes) implements DistributedMsg {}

    // ── State ────────────────────────────────────────────────────────────────

    record DistributedState(
        String leaderNodeId,
        List<String> replicaNodes,
        long currentVersion,
        ConcurrentHashMap<String, Object> sharedState
    ) {}

    // ── Leader Election Process ───────────────────────────────────────────────

    public static Proc<DistributedState, DistributedMsg> createLeaderElection(
        String localNodeId,
        List<String> allNodes
    ) {
        return new Proc<>(
            new DistributedState(null, allNodes, 0, new ConcurrentHashMap<>()),
            (state, msg) -> switch (msg) {
                case ElectLeader _ -> {
                    // Simple deterministic leader election (lowest node ID wins)
                    var leader = allNodes.stream().min(String::compareTo).orElse();

                    System.out.println("[LeaderElection] Elected leader: " + leader);

                    yield new DistributedState(
                        leader,
                        allNodes.stream().filter(n -> !n.equals(leader)).toList(),
                        state.currentVersion(),
                        state.sharedState()
                    );
                }

                case NodeJoin(var nodeId, var host, var port) -> {
                    System.out.println("[LeaderElection] Node joined: " + nodeId);

                    var newNodes = new java.util.ArrayList<>(allNodes);
                    newNodes.add(nodeId);

                    yield new DistributedState(
                        state.leaderNodeId(),
                        newNodes.stream().filter(n -> !n.equals(state.leaderNodeId())).toList(),
                        state.currentVersion(),
                        state.sharedState()
                    );
                }

                case ReplicateState(var key, var value) -> {
                    if (localNodeId.equals(state.leaderNodeId())) {
                        // Leader: Update state and trigger replication
                        state.sharedState().put(key, value);

                        // Replicate to all replicas
                        for (var replica : state.replicaNodes()) {
                            // Send replication message via gRPC
                            System.out.println("[Leader] Replicating to " + replica + ": " + key);
                        }

                        yield new DistributedState(
                            state.leaderNodeId(),
                            state.replicaNodes(),
                            state.currentVersion() + 1,
                            state.sharedState()
                        );
                    } else {
                        // Replica: Forward to leader
                        System.out.println("[Replica] Forwarding to leader: " + state.leaderNodeId());

                        yield state;
                    }
                }

                case StateUpdate(var key, var value, var version) -> {
                    // Replica: Apply state update from leader
                    if (version > state.currentVersion()) {
                        state.sharedState().put(key, value);

                        System.out.println("[Replica] Applied state update: " + key + " = " + value);

                        yield new DistributedState(
                            state.leaderNodeId(),
                            state.replicaNodes(),
                            version,
                            state.sharedState()
                        );
                    } else {
                        yield state;
                    }
                }

                default -> state;
            }
        );
    }

    // ── Usage Example ────────────────────────────────────────────────────────

    public static void main(String[] args) throws Exception {
        var nodes = List.of("node-1", "node-2", "node-3");

        var node1 = createLeaderElection("node-1", nodes);
        var node2 = createLeaderElection("node-2", nodes);
        var node3 = createLeaderElection("node-3", nodes);

        // Trigger leader election
        node1.tell(new ElectLeader());
        Thread.sleep(100);

        node2.tell(new ElectLeader());
        Thread.sleep(100);

        node3.tell(new ElectLeader());
        Thread.sleep(100);

        // Replicate state
        node1.tell(new ReplicateState("config", "production"));
        Thread.sleep(100);

        node2.tell(new ReplicateState("config", "staging"));
        Thread.sleep(100);

        node1.stop();
        node2.stop();
        node3.stop();
    }
}
```

```java
    commitState();
}
```

## Part 3: Consensus and Coordination

### Raft Implementation for JOTP

```java
package io.github.seanchatmangpt.jotp.examples.distributed;

import io.github.seanchatmangpt.jotp.*;
import java.time.Duration;
import java.util.*;
import java.util.concurrent.*;

/**
 * JotpRaft - Raft consensus algorithm implementation for JOTP
 * distributed coordination.
 */
public class JotpRaft {

    // ── Message Types ────────────────────────────────────────────────────────

    sealed interface RaftMsg permits
        RequestVote,
        VoteResponse,
        AppendEntries,
        AppendResponse,
        ClientRequest {}

    record RequestVote(
        int term,
        String candidateId,
        int lastLogIndex,
        int lastLogTerm
    ) implements RaftMsg {}

    record VoteResponse(
        int term,
        boolean voteGranted
    ) implements RaftMsg {}

    record AppendEntries(
        int term,
        String leaderId,
        int prevLogIndex,
        int prevLogTerm,
        List<LogEntry> entries,
        int leaderCommit
    ) implements RaftMsg {}

    record AppendResponse(
        int term,
        boolean success
    ) implements RaftMsg {}

    record ClientRequest(String command) implements RaftMsg {}

    // ── State ────────────────────────────────────────────────────────────────

    record RaftState(
        String nodeId,
        ServerRole role,
        int currentTerm,
        String votedFor,
        List<LogEntry> log,
        int commitIndex,
        int lastApplied,
        Map<String, Integer> nextIndex,
        Map<String, Integer> matchIndex
    ) {}

    enum ServerRole { FOLLOWER, CANDIDATE, LEADER }

    record LogEntry(int term, String command) {}

    // ── Raft Process ─────────────────────────────────────────────────────────

    public static Proc<RaftState, RaftMsg> createRaftNode(
        String nodeId,
        List<String> allNodes,
        Duration electionTimeout
    ) {
        var randomTimeout = Duration.ofMillis(
            electionTimeout.toMillis() + ThreadLocalRandom.current().nextLong(500)
        );

        return new Proc<>(
            new RaftState(
                nodeId,
                ServerRole.FOLLOWER,
                0,
                null,
                new ArrayList<>(),
                0,
                0,
                new HashMap<>(),
                new HashMap<>()
            ),
            (state, msg) -> switch (msg) {
                case RequestVote(var term, var candidateId, var lastLogIndex, var lastLogTerm) -> {
                    System.out.printf("[%s] Received vote request from %s (term %d)%n", nodeId, candidateId, term);

                    boolean voteGranted = false;

                    if (term > state.currentTerm()) {
                        // Vote for candidate
                        voteGranted = true;

                        // Reset election timer
                        // resetElectionTimer();

                        yield new RaftState(
                            state.nodeId(),
                            ServerRole.FOLLOWER,
                            term,
                            candidateId,
                            state.log(),
                            state.commitIndex(),
                            state.lastApplied(),
                            state.nextIndex(),
                            state.matchIndex()
                        );
                    } else {
                        yield state;
                    }
                }

                case VoteResponse(var term, var voteGranted) -> {
                    if (state.role() == ServerRole.CANDIDATE && term == state.currentTerm()) {
                        if (voteGranted) {
                            // Count votes
                            // If majority reached, become leader
                            System.out.printf("[%s] Received vote (term %d)%n", nodeId, term);

                            yield new RaftState(
                                state.nodeId(),
                                ServerRole.LEADER,
                                state.currentTerm(),
                                state.votedFor(),
                                state.log(),
                                state.commitIndex(),
                                state.lastApplied(),
                                state.nextIndex(),
                                state.matchIndex()
                            );
                        } else {
                            yield state;
                        }
                    } else {
                        yield state;
                    }
                }

                case AppendEntries(var term, var leaderId, var prevLogIndex, var prevLogTerm, var entries, var leaderCommit) -> {
                    System.out.printf("[%s] Received append entries from %s (term %d)%n", nodeId, leaderId, term);

                    if (term >= state.currentTerm()) {
                        // Accept entries and update commit index
                        var newLog = new ArrayList<>(state.log());
                        newLog.addAll(entries);

                        yield new RaftState(
                            state.nodeId(),
                            ServerRole.FOLLOWER,
                            term,
                            state.votedFor(),
                            newLog,
                            Math.min(leaderCommit, newLog.size() - 1),
                            state.lastApplied(),
                            state.nextIndex(),
                            state.matchIndex()
                        );
                    } else {
                        yield state;
                    }
                }

                case ClientRequest(var command) -> {
                    if (state.role() == ServerRole.LEADER) {
                        System.out.printf("[%s] Leader appending log entry: %s%n", nodeId, command);

                        var newLog = new ArrayList<>(state.log());
                        newLog.add(new LogEntry(state.currentTerm(), command));

                        // Replicate to followers
                        // replicateEntries(newLog);

                        yield new RaftState(
                            state.nodeId(),
                            ServerRole.LEADER,
                            state.currentTerm(),
                            state.votedFor(),
                            newLog,
                            state.commitIndex(),
                            state.lastApplied(),
                            state.nextIndex(),
                            state.matchIndex()
                        );
                    } else {
                        // Forward to leader
                        System.out.println("[%s] Forwarding request to leader");
                        yield state;
                    }
                }

                default -> state;
            }
        );
    }

    public static void main(String[] args) throws Exception {
        var nodes = List.of("node-1", "node-2", "node-3");

        var node1 = createRaftNode("node-1", nodes, Duration.ofSeconds(5));
        var node2 = createRaftNode("node-2", nodes, Duration.ofSeconds(5));
        var node3 = createRaftNode("node-3", nodes, Duration.ofSeconds(5));

        // Simulate election
        node1.tell(new RequestVote(1, "node-1", 0, 0));
        Thread.sleep(100);

        node2.tell(new VoteResponse(1, true));
        node3.tell(new VoteResponse(1, true));

        Thread.sleep(100);

        // Leader commits log entry
        node1.tell(new ClientRequest("SET x = 42"));

        Thread.sleep(500);

        node1.stop();
        node2.stop();
        node3.stop();
    }
}
```

## Common Pitfalls

### 1. Ignoring Network Latency

<Callout type="error">
**Don't assume zero-latency communication**

```java
// BAD: Synchronous ask() across nodes
var response = remoteProc.ask(new Request(), Duration.ofSeconds(1));
// Network delay = timeout!

// GOOD: Use tell() for fire-and-forget, handle async responses
remoteProc.tell(new Request());
```
</Callout>

### 2. No Failure Detection

<Callout type="error">
**Implement heartbeat-based failure detection**

Without failure detection, your system will route traffic to dead nodes.
</Callout>

### 3. Split-Brain Scenarios

<Callout type="error">
**Use quorum-based decision making**

Never allow a minority partition to make decisions unilaterally.
</Callout>

```java
    var state1 = node1.ask(new GetState(), 100).get();
    var state2 = node2.ask(new GetState(), 100).get();
    var state3 = node3.ask(new GetState(), 100).get();

    assertEquals(state1, state2);
    assertEquals(state2, state3);

    node1.stop();
    node2.stop();
    node3.stop();
}
```
