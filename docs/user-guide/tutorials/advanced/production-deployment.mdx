## Part 1: Monitoring and Observability

### Prometheus Metrics Integration

```java
package io.github.seanchatmangpt.jotp.examples.production;

import io.github.seanchatmangpt.jotp.*;
import io.prometheus.client.CollectorRegistry;
import io.prometheus.client.Counter;
import io.prometheus.client.Histogram;
import io.prometheus.client.Gauge;
import io.prometheus.client.exporter.HTTPServer;
import java.io.IOException;

/**
 * ProductionMonitoring - Prometheus metrics integration for
 * production observability of JOTP applications.
 */
public class ProductionMonitoring {

    // ── Metrics Registry ─────────────────────────────────────────────────────

    private static final CollectorRegistry registry = new CollectorRegistry(true);

    // ── Process Metrics ──────────────────────────────────────────────────────

    private static final Counter messagesProcessed = Counter.build()
        .name("jotp_messages_processed_total")
        .help("Total messages processed")
        .labelNames("process_type", "process_id")
        .register(registry);

    private static final Counter messagesFailed = Counter.build()
        .name("jotp_messages_failed_total")
        .help("Total messages failed")
        .labelNames("process_type", "process_id", "error_type")
        .register(registry);

    private static final Histogram messageLatency = Histogram.build()
        .name("jotp_message_latency_seconds")
        .help("Message processing latency")
        .labelNames("process_type", "process_id")
        .buckets(0.001, 0.005, 0.01, 0.025, 0.05, 0.1, 0.25, 0.5, 1.0, 2.5, 5.0, 10.0)
        .register(registry);

    // ── Supervisor Metrics ──────────────────────────────────────────────────

    private static final Counter childRestarts = Counter.build()
        .name("jotp_supervisor_child_restarts_total")
        .help("Total child process restarts")
        .labelNames("supervisor_id", "child_id")
        .register(registry);

    private static final Gauge activeChildren = Gauge.build()
        .name("jotp_supervisor_active_children")
        .help("Number of active child processes")
        .labelNames("supervisor_id")
        .register(registry);

    // ── System Metrics ───────────────────────────────────────────────────────

    private static final Gauge processCount = Gauge.build()
        .name("jotp_process_count")
        .help("Total number of processes")
        .register(registry);

    private static final Gauge virtualThreadCount = Gauge.build()
        .name("jotp_virtual_thread_count")
        .help("Number of virtual threads")
        .register(registry);

    // ── Instrumented Process ────────────────────────────────────────────────

    public static Proc<Long, String> createInstrumentedProcess(
        String processId,
        String processType
    ) {
        return new Proc<>(
            0L,
            (state, msg) -> {
                var start = System.nanoTime();

                try {
                    // Process message
                    var newState = state + 1;

                    // Record success metrics
                    messagesProcessed.labels(processType, processId).inc();
                    messageLatency.labels(processType, processId)
                        .observe((System.nanoTime() - start) / 1e9);

                    return newState;
                } catch (Exception e) {
                    // Record failure metrics
                    messagesFailed.labels(processType, processId, e.getClass().getSimpleName()).inc();
                    messageLatency.labels(processType, processId)
                        .observe((System.nanoTime() - start) / 1e9);
                    throw e;
                }
            }
        );
    }

    // ── Instrumented Supervisor ──────────────────────────────────────────────

    public static Supervisor createInstrumentedSupervisor(
        String supervisorId,
        Supervisor.Strategy strategy,
        int maxRestarts,
        java.time.Duration restartWindow
    ) {
        var supervisor = new Supervisor(strategy, maxRestarts, restartWindow);

        // Add metrics listener
        supervisor.addHealthListener(event -> {
            if (event instanceof SupervisorEvent.ChildRestarted restart) {
                childRestarts.labels(supervisorId, restart.childId()).inc();
            }

            if (event instanceof SupervisorEvent.ChildStarted) {
                activeChildren.labels(supervisorId).inc();
            }

            if (event instanceof SupervisorEvent.ChildStopped) {
                activeChildren.labels(supervisorId).dec();
            }
        });

        return supervisor;
    }

    // ── Metrics Server ───────────────────────────────────────────────────────

    private static HTTPServer metricsServer;

    public static void startMetricsServer(int port) throws IOException {
        metricsServer = new HTTPServer(port, registry);
        System.out.println("[METRICS] Prometheus metrics server started on port " + port);
    }

    public static void stopMetricsServer() {
        if (metricsServer != null) {
            metricsServer.stop();
        }
    }

    // ── Usage Example ────────────────────────────────────────────────────────

    public static void main(String[] args) throws Exception {
        // Start metrics server
        startMetricsServer(9090);

        // Create instrumented supervisor
        var supervisor = createInstrumentedSupervisor(
            "main-supervisor",
            Supervisor.Strategy.ONE_FOR_ONE,
            5,
            java.time.Duration.ofSeconds(60)
        );

        // Spawn instrumented processes
        for (int i = 0; i < 10; i++) {
            var processId = "worker-" + i;
            supervisor.supervise(
                processId,
                0L,
                createInstrumentedProcess(processId, "worker")
            );
        }

        // Update process count gauge
        processCount.set(10);
        virtualThreadCount.set(Thread.activeCount());

        System.out.println("[PRODUCTION] Monitoring ready");
        System.out.println("[PRODUCTION] Metrics available at http://localhost:9090/metrics");

        // Keep running
        Thread.sleep(java.time.Duration.ofMinutes(5).toMillis());

        // Cleanup
        supervisor.shutdown();
        stopMetricsServer();
    }
}
```

```yaml

groups:
  - name: jotp_alerts
    interval: 30s
    rules:
      - alert: HighMessageFailureRate
        expr: |
          rate(jotp_messages_failed_total[5m]) /
          rate(jotp_messages_processed_total[5m]) > 0.05
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "High message failure rate detected"
          description: "Message failure rate is above 5% for 5 minutes"

      - alert: SupervisorRestartStorm
        expr: |
          rate(jotp_supervisor_child_restarts_total[1m]) > 10
        for: 2m
        labels:
          severity: critical
        annotations:
          summary: "Supervisor restart storm detected"
          description: "More than 10 child restarts per minute"

      - alert: HighProcessCount
        expr: jotp_process_count > 100000
        for: 5m
```

```java
    public static <S, M> Proc<S, M> createTracedProcess(
        String processName,
        S initialState,
        java.util.function.BiFunction<S, M, S> handler
    ) {
        return new Proc<>(
            initialState,
            (state, msg) -> {
                var spanName = processName + "." + msg.getClass().getSimpleName();

                var span = tracer.spanBuilder(spanName)
                    .setParent(Context.current())
                    .startSpan();

                try (var scope = span.makeCurrent()) {
                    // Add message attributes
                    span.setAttribute("jotp.process", processName);
                    span.setAttribute("jotp.message_type", msg.getClass().getSimpleName());

                    // Process message
                    var newState = handler.apply(state, msg);

                    span.setStatus(io.opentelemetry.api.trace.StatusCode.OK);

                    return newState;
                } catch (Exception e) {
                    span.recordException(e);
                    span.setStatus(io.opentelemetry.api.trace.StatusCode.ERROR, e.getMessage());
                    throw e;
                } finally {
                    span.end();
                }
            }
        );
    }

        String machineName,
        StateMachine<S, E, D> machine
    ) {
        // Wrap state machine to trace transitions
        return machine;  // Simplified - actual implementation would wrap transitions
    }

    // ── Usage Example ────────────────────────────────────────────────────────

    public static void main(String[] args) throws Exception {
```

```java
package io.github.seanchatmangpt.jotp.examples.production;

import io.github.seanchatmangpt.jotp.*;
import com.sun.net.httpserver.HttpServer;
import com.sun.net.httpserver.HttpHandler;

/**
 * HealthCheckServer - Kubernetes health check endpoints for
 * JOTP applications.
 */
public class HealthCheckServer {

    private HttpServer server;
    private Supervisor mainSupervisor;

    public HealthCheckServer(int port, Supervisor supervisor) throws Exception {
        this.mainSupervisor = supervisor;
        this.server = HttpServer.create(new java.net.InetSocketAddress(port), 0);

        // Liveness probe
        server.createContext("/health/live", exchange -> {
            var response = "OK";
            exchange.sendResponseHeaders(200, response.length());
            exchange.getResponseBody().write(response.getBytes());
            exchange.close();
        });

        // Readiness probe
        server.createContext("/health/ready", exchange -> {
            if (isReady()) {
                var response = "Ready";
                exchange.sendResponseHeaders(200, response.length());
                exchange.getResponseBody().write(response.getBytes());
            } else {
                var response = "Not Ready";
                exchange.sendResponseHeaders(503, response.length());
                exchange.getResponseBody().write(response.getBytes());
            }
            exchange.close();
        });

        // Startup probe
        server.createContext("/health/startup", exchange -> {
            if (isStarted()) {
                var response = "Started";
                exchange.sendResponseHeaders(200, response.length());
                exchange.getResponseBody().write(response.getBytes());
            } else {
                var response = "Starting";
                exchange.sendResponseHeaders(503, response.length());
                exchange.getResponseBody().write(response.getBytes());
            }
            exchange.close();
        });

        server.start();
        System.out.println("[HEALTH] Health check server started on port " + port);
    }

    private boolean isReady() {
        // Check if supervisor is running
        if (mainSupervisor == null || !mainSupervisor.isRunning()) {
            return false;
        }

        // Check if minimum children are ready
        // (custom logic based on your application)

        return true;
    }

    private boolean isStarted() {
        // Check if application has completed startup
        return mainSupervisor != null && mainSupervisor.isRunning();
    }

    public void stop() {
        server.stop(0);
    }

    // ── Usage Example ────────────────────────────────────────────────────────
```

## Part 5: Scaling Strategies

### Horizontal Pod Autoscaling

```yaml
# hpa.yaml
apiVersion: autoscaling/v2
kind: HorizontalPodAutoscaler
metadata:
  name: jotp-hpa
spec:
  scaleTargetRef:
    apiVersion: apps/v1
    kind: Deployment
    name: jotp-application
  minReplicas: 3
  maxReplicas: 10
  metrics:
  - type: Resource
    resource:
      name: cpu
      target:
        type: Utilization
        averageUtilization: 70
  - type: Resource
    resource:
      name: memory
      target:
        type: Utilization
        averageUtilization: 80
  - type: Pods
    pods:
      metric:
        name: jotp_messages_processed_per_second
      target:
        type: AverageValue
        averageValue: "25000"
  behavior:
    scaleDown:
      stabilizationWindowSeconds: 300
      policies:
      - type: Percent
        value: 50
        periodSeconds: 60
    scaleUp:
      stabilizationWindowSeconds: 0
      policies:
      - type: Percent
        value: 100
        periodSeconds: 30
      - type: Pods
        value: 2
        periodSeconds: 30
      selectPolicy: Max
```

<Callout type="info">
**Pro Tip: Scaling Thresholds**

Based on production benchmarks:
- **CPU**: Scale at 70% utilization (leaves headroom for spikes)
- **Memory**: Scale at 80% utilization (avoid GC pressure)
- **Throughput**: Scale when >25K ops/sec per pod (conservative)
- **Latency**: Scale when p99 >50ms for 5 minutes

Adjust based on your SLA requirements.
</Callout>
