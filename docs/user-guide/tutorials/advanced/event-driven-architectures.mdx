## Part 1: Event Manager Patterns

### Pattern: Hierarchical Event Distribution

```java
package io.github.seanchatmangpt.jotp.examples.events;

import io.github.seanchatmangpt.jotp.*;
import java.util.ArrayList;
import java.util.List;
import java.util.concurrent.CopyOnWriteArrayList;

/**
 * HierarchicalEventManager - Demonstrates event distribution
 * across multiple layers with filtering and transformation.
 */
public class HierarchicalEventManager {

    // ── Event Types ──────────────────────────────────────────────────────────

    sealed interface DomainEvent permits
        OrderEvent,
        InventoryEvent,
        PaymentEvent,
        ShippingEvent {}

    sealed interface OrderEvent extends DomainEvent permits
        OrderCreated,
        OrderUpdated,
        OrderCancelled {}

    sealed interface InventoryEvent extends DomainEvent permits
        InventoryReserved,
        InventoryReleased,
        StockLow {}

    sealed interface PaymentEvent extends DomainEvent permits
        PaymentAuthorized,
        PaymentCaptured,
        PaymentDeclined {}

    sealed interface ShippingEvent extends DomainEvent permits
        OrderShipped,
        OrderDelivered {}

    record OrderCreated(String orderId, String customerId, String items) implements OrderEvent {}
    record OrderUpdated(String orderId, String updates) implements OrderEvent {}
    record OrderCancelled(String orderId, String reason) implements OrderEvent {}

    record InventoryReserved(String orderId, String items) implements InventoryEvent {}
    record InventoryReleased(String orderId, String items) implements InventoryEvent {}
    record StockLow(String productId, int remaining) implements InventoryEvent {}

    record PaymentAuthorized(String orderId, double amount, String authCode) implements PaymentEvent {}
    record PaymentCaptured(String orderId, String transactionId) implements PaymentEvent {}
    record PaymentDeclined(String orderId, String reason) implements PaymentEvent {}

    record OrderShipped(String orderId, String trackingNumber) implements ShippingEvent {}
    record OrderDelivered(String orderId, String signature) implements ShippingEvent {}

    // ── Event Manager ────────────────────────────────────────────────────────

    public static class EventManager<E> {
        private final List<ProcRef<Object, E>> subscribers = new CopyOnWriteArrayList<>();
        private final List<EventHandler<E>> handlers = new CopyOnWriteArrayList<>();

        public interface EventHandler<E> {
            void handle(E event);
            boolean canHandle(E event);
        }

        public void subscribe(ProcRef<Object, E> subscriber) {
            subscribers.add(subscriber);
        }

        public void subscribeHandler(EventHandler<E> handler) {
            handlers.add(handler);
        }

        public void publish(E event) {
            // Notify process subscribers
            for (var subscriber : subscribers) {
                try {
                    subscriber.tell(event);
                } catch (Exception e) {
                    System.err.println("Failed to notify subscriber: " + e.getMessage());
                }
            }

            // Notify handler subscribers
            for (var handler : handlers) {
                if (handler.canHandle(event)) {
                    try {
                        handler.handle(event);
                    } catch (Exception e) {
                        System.err.println("Failed to handle event: " + e.getMessage());
                    }
                }
            }
        }

        public void unsubscribe(ProcRef<Object, E> subscriber) {
            subscribers.remove(subscriber);
        }

        public int subscriberCount() {
            return subscribers.size() + handlers.size();
        }
    }

    // ── Filtered Event Handler ───────────────────────────────────────────────

    public static abstract class FilteredEventHandler<E> implements EventManager.EventHandler<E> {
        private final Class<E> eventType;

        protected FilteredEventHandler(Class<E> eventType) {
            this.eventType = eventType;
        }

        @Override
        public boolean canHandle(E event) {
            return eventType.isInstance(event);
        }

        @Override
        public void handle(E event) {
            if (canHandle(event)) {
                onEvent(event);
            }
        }

        protected abstract void onEvent(E event);
    }

    // ── Projection Process ───────────────────────────────────────────────────

    public static class ProjectionProcess {
        public static ProcRef<List<DomainEvent>, DomainEvent> create(
            String name,
            EventManager<DomainEvent> eventManager
        ) {
            var events = new ArrayList<DomainEvent>();

            var process = Proc.spawn(
                events,
                (state, event) -> {
                    System.out.printf("[%s] Received event: %s%n", name, event);

                    var newEvents = new ArrayList<>(state);
                    newEvents.add(event);

                    return newEvents;
                }
            );

            eventManager.subscribe(ProcRef.<List<DomainEvent>, DomainEvent>self());

            return ProcRef.<List<DomainEvent>, DomainEvent>self();
        }
    }

    // ── Usage Example ────────────────────────────────────────────────────────

    public static void main(String[] args) throws Exception {
        var eventManager = new EventManager<DomainEvent>();

        // Create projections
        var readModel = ProjectionProcess.create("ReadModel", eventManager);
        var analytics = ProjectionProcess.create("Analytics", eventManager);
        var notifications = ProjectionProcess.create("Notifications", eventManager);

        // Add filtered handler for order events
        eventManager.subscribeHandler(new FilteredEventHandler<OrderEvent>(OrderEvent.class) {
            @Override
            protected void onEvent(OrderEvent event) {
                System.out.println("[ORDER HANDLER] Processing order event: " + event);
            }
        });

        // Publish events
        System.out.println("=== Publishing Events ===\n");

        eventManager.publish(new OrderCreated("order-123", "customer-456", "item-1,item-2"));
        Thread.sleep(50);

        eventManager.publish(new InventoryReserved("order-123", "item-1,item-2"));
        Thread.sleep(50);

        eventManager.publish(new PaymentAuthorized("order-123", 99.99, "AUTH-789"));
        Thread.sleep(50);

        eventManager.publish(new OrderShipped("order-123", "TRACK-ABC"));
        Thread.sleep(50);

        System.out.printf("\nTotal subscribers: %d%n", eventManager.subscriberCount());

        Thread.sleep(100);
    }
}
```

## Part 2: CQRS Implementation

### Pattern: Command-Query Separation

```java
package io.github.seanchatmangpt.jotp.examples.events;

import io.github.seanchatmangpt.jotp.*;
import java.util.ArrayList;
import java.util.List;
import java.util.concurrent.ConcurrentHashMap;

/**
 * CQRSPattern - Command Query Responsibility Segregation with
 * separate read and write models.
 */
public class CQRSPattern {

    // ── Commands (Write Side) ────────────────────────────────────────────────

    sealed interface Command permits
        CreateOrder,
        UpdateOrder,
        CancelOrder {}

    record CreateOrder(
        String orderId,
        String customerId,
        String items,
        double amount
    ) implements Command {}

    record UpdateOrder(
        String orderId,
        String updates
    ) implements Command {}

    record CancelOrder(
        String orderId,
        String reason
    ) implements Command {}

    // ── Queries (Read Side) ─────────────────────────────────────────────────

    sealed interface Query<T> permits
        GetOrder,
        GetCustomerOrders,
        GetOrderStats {}

    record GetOrder(String orderId) implements Query<OrderDto> {}
    record GetCustomerOrders(String customerId) implements Query<List<OrderDto>> {}
    record GetOrderStats() implements Query<OrderStats> {}

    // ── Events (Bridge) ──────────────────────────────────────────────────────

    sealed interface OrderEvent permits
        OrderCreatedEvent,
        OrderUpdatedEvent,
        OrderCancelledEvent {}

    record OrderCreatedEvent(
        String orderId,
        String customerId,
        String items,
        double amount,
        long timestamp
    ) implements OrderEvent {}

    record OrderUpdatedEvent(
        String orderId,
        String updates,
        long timestamp
    ) implements OrderEvent {}

    record OrderCancelledEvent(
        String orderId,
        String reason,
        long timestamp
    ) implements OrderEvent {}

    // ── DTOs (Read Model) ───────────────────────────────────────────────────

    record OrderDto(
        String orderId,
        String customerId,
        String items,
        double amount,
        String status,
        long updatedAt
    ) {}

    record OrderStats(
        int totalOrders,
        int activeOrders,
        int cancelledOrders,
        double totalRevenue
    ) {}

    // ── Write Side: Command Handler ──────────────────────────────────────────

    public static class CommandHandler {
        private final EventManager<OrderEvent> eventManager;

        public CommandHandler(EventManager<OrderEvent> eventManager) {
            this.eventManager = eventManager;
        }

        public void handle(Command command) {
            switch (command) {
                case CreateOrder(var orderId, var customerId, var items, var amount) -> {
                    // Validate command
                    validateOrder(items, amount);

                    // Emit event
                    var event = new OrderCreatedEvent(
                        orderId,
                        customerId,
                        items,
                        amount,
                        System.currentTimeMillis()
                    );

                    eventManager.publish(event);
                }

                case UpdateOrder(var orderId, var updates) -> {
                    var event = new OrderUpdatedEvent(
                        orderId,
                        updates,
                        System.currentTimeMillis()
                    );

                    eventManager.publish(event);
                }

                case CancelOrder(var orderId, var reason) -> {
                    var event = new OrderCancelledEvent(
                        orderId,
                        reason,
                        System.currentTimeMillis()
                    );

                    eventManager.publish(event);
                }
            }
        }

        private void validateOrder(String items, double amount) {
            if (items == null || items.isBlank()) {
                throw new IllegalArgumentException("Items cannot be empty");
            }
            if (amount <= 0) {
                throw new IllegalArgumentException("Amount must be positive");
            }
        }
    }

    // ── Read Side: Projection Builder ────────────────────────────────────────

    public static class ReadModelProjection {
        private final ConcurrentHashMap<String, OrderDto> orders = new ConcurrentHashMap<>();
        private final ConcurrentHashMap<String, List<String>> customerOrders = new ConcurrentHashMap<>();
        private int totalOrders = 0;
        private int cancelledOrders = 0;
        private double totalRevenue = 0.0;

        public void handle(OrderEvent event) {
            switch (event) {
                case OrderCreatedEvent(var orderId, var customerId, var items, var amount, var timestamp) -> {
                    var order = new OrderDto(
                        orderId,
                        customerId,
                        items,
                        amount,
                        "ACTIVE",
                        timestamp
                    );

                    orders.put(orderId, order);
                    customerOrders.computeIfAbsent(customerId, k -> new ArrayList<>()).add(orderId);
                    totalOrders++;
                    totalRevenue += amount;

                    System.out.println("[READ MODEL] Order created: " + orderId);
                }

                case OrderUpdatedEvent(var orderId, var updates, var timestamp) -> {
                    var existing = orders.get(orderId);
                    if (existing != null) {
                        var updated = new OrderDto(
                            existing.orderId(),
                            existing.customerId(),
                            existing.items(),
                            existing.amount(),
                            "UPDATED",
                            timestamp
                        );

                        orders.put(orderId, updated);
                        System.out.println("[READ MODEL] Order updated: " + orderId);
                    }
                }

                case OrderCancelledEvent(var orderId, var reason, var timestamp) -> {
                    var existing = orders.get(orderId);
                    if (existing != null) {
                        var cancelled = new OrderDto(
                            existing.orderId(),
                            existing.customerId(),
                            existing.items(),
                            existing.amount(),
                            "CANCELLED",
                            timestamp
                        );

                        orders.put(orderId, cancelled);
                        cancelledOrders++;

                        System.out.println("[READ MODEL] Order cancelled: " + orderId);
                    }
                }
            }
        }

        public OrderDto getOrder(String orderId) {
            return orders.get(orderId);
        }

        public List<OrderDto> getCustomerOrders(String customerId) {
            var orderIds = customerOrders.getOrDefault(customerId, List.of());
            return orderIds.stream()
                .map(orders::get)
                .toList();
        }

        public OrderStats getStats() {
            return new OrderStats(
                totalOrders,
                totalOrders - cancelledOrders,
                cancelledOrders,
                totalRevenue
            );
        }
    }

    // ── Query Handler ───────────────────────────────────────────────────────

    public static class QueryHandler {
        private final ReadModelProjection readModel;

        public QueryHandler(ReadModelProjection readModel) {
            this.readModel = readModel;
        }

        @SuppressWarnings("unchecked")
        public <T> T handle(Query<T> query) {
            return switch (query) {
                case GetOrder(var orderId) -> (T) readModel.getOrder(orderId);
                case GetCustomerOrders(var customerId) -> (T) readModel.getCustomerOrders(customerId);
                case GetOrderStats _ -> (T) readModel.getStats();
            };
        }
    }

    // ── Usage Example ────────────────────────────────────────────────────────

    public static void main(String[] args) throws Exception {
        var eventManager = new EventManager<OrderEvent>();
        var readModel = new ReadModelProjection();
        var commandHandler = new CommandHandler(eventManager);
        var queryHandler = new QueryHandler(readModel);

        // Subscribe read model to events
        eventManager.subscribeHandler(new EventManager.EventHandler<OrderEvent>() {
            @Override
            public void handle(OrderEvent event) {
                readModel.handle(event);
            }

            @Override
            public boolean canHandle(OrderEvent event) {
                return true;
            }
        });

        // ── Command Side (Write) ─────────────────────────────────────────────

        System.out.println("=== Command Side (Write) ===\n");

        commandHandler.handle(new CreateOrder("order-1", "customer-123", "item-1,item-2", 99.99));
        Thread.sleep(50);

        commandHandler.handle(new CreateOrder("order-2", "customer-123", "item-3", 149.99));
        Thread.sleep(50);

        commandHandler.handle(new UpdateOrder("order-1", " Expedited shipping"));
        Thread.sleep(50);

        commandHandler.handle(new CancelOrder("order-2", "Customer request"));
        Thread.sleep(50);

        // ── Query Side (Read) ────────────────────────────────────────────────

        System.out.println("\n=== Query Side (Read) ===\n");

        var order = queryHandler.handle(new GetOrder("order-1"));
        System.out.println("Order: " + order);

        var customerOrders = queryHandler.handle(new GetCustomerOrders("customer-123"));
        System.out.println("Customer orders: " + customerOrders.size() + " orders");

        var stats = queryHandler.handle(new GetOrderStats());
        System.out.printf("Stats: %d total, %d active, %d cancelled, $%.2f revenue%n",
            stats.totalOrders(), stats.activeOrders(), stats.cancelledOrders(), stats.totalRevenue());

        Thread.sleep(100);
    }
}
```

<Callout type="info">
**Pro Tip: CQRS Benefits**

1. **Scalability**: Read and write models can scale independently
2. **Performance**: Read queries don't block write commands
3. **Flexibility**: Different read models for different use cases
4. **Audit Trail**: Event log provides complete history
5. **Event Replay**: Rebuild read models from event log
</Callout>

## Part 3: Event Sourcing

### Pattern: Event Store and Replay

```java
package io.github.seanchatmangpt.jotp.examples.events;

import io.github.seanchatmangpt.jotp.*;
import java.util.ArrayList;
import java.util.List;
import java.util.concurrent.CopyOnWriteArrayList;

/**
 * EventSourcing - Complete event sourcing implementation with
 * event store, snapshotting, and replay capabilities.
 */
public class EventSourcing {

    // ── Event Store ─────────────────────────────────────────────────────────

    public static class EventStore<E> {
        private final List<StoredEvent<E>> events = new CopyOnWriteArrayList<>();

        public void append(String streamId, E event, long version) {
            var stored = new StoredEvent<>(
                events.size(),
                streamId,
                version,
                event,
                System.currentTimeMillis()
            );

            events.add(stored);
            System.out.println("[EVENT STORE] Appended: " + streamId + " v" + version);
        }

        public List<StoredEvent<E>> getStream(String streamId) {
            return events.stream()
                .filter(e -> e.streamId().equals(streamId))
                .toList();
        }

        public List<StoredEvent<E>> getStreamFromVersion(String streamId, long fromVersion) {
            return events.stream()
                .filter(e -> e.streamId().equals(streamId) && e.version() >= fromVersion)
                .toList();
        }

        public List<StoredEvent<E>> getAllEvents() {
            return new ArrayList<>(events);
        }

        public long getCurrentVersion(String streamId) {
            return events.stream()
                .filter(e -> e.streamId().equals(streamId))
                .mapToLong(StoredEvent::version)
                .max()
                .orElse(0);
        }
    }

    record StoredEvent<E>(
        long eventNumber,
        String streamId,
        long version,
        event E,
        long timestamp
    ) {}

    // ── Aggregate (Event Sourced Entity) ─────────────────────────────────────

    public static class OrderAggregate {
        private final String orderId;
        private final List<OrderEvent> history = new ArrayList<>();
        private final EventStore<OrderEvent> eventStore;

        private String customerId;
        private String items;
        private double amount;
        private String status;
        private long version = 0;

        public OrderAggregate(String orderId, EventStore<OrderEvent> eventStore) {
            this.orderId = orderId;
            this.eventStore = eventStore;
        }

        public void loadFromHistory() {
            var events = eventStore.getStream(orderId);

            for (var storedEvent : events) {
                apply(storedEvent.event());
                this.version = storedEvent.version();
            }

            System.out.println("[AGGREGATE] Loaded " + events.size() + " events for " + orderId);
        }

        public void replayFromVersion(long fromVersion) {
            var events = eventStore.getStreamFromVersion(orderId, fromVersion);

            for (var storedEvent : events) {
                apply(storedEvent.event());
                this.version = storedEvent.version();
            }

            System.out.println("[AGGREGATE] Replayed " + events.size() + " events from v" + fromVersion);
        }

        public void createOrder(String customerId, String items, double amount) {
            if (version > 0) {
                throw new IllegalStateException("Order already exists");
            }

            var event = new OrderCreatedEvent(
                orderId,
                customerId,
                items,
                amount,
                System.currentTimeMillis()
            );

            applyAndPersist(event);
        }

        public void cancelOrder(String reason) {
            if (!"ACTIVE".equals(status)) {
                throw new IllegalStateException("Order not active");
            }

            var event = new OrderCancelledEvent(
                orderId,
                reason,
                System.currentTimeMillis()
            );

            applyAndPersist(event);
        }

        private void apply(OrderEvent event) {
            switch (event) {
                case OrderCreatedEvent(var oid, var cid, var it, var amt, var ts) -> {
                    this.customerId = cid;
                    this.items = it;
                    this.amount = amt;
                    this.status = "ACTIVE";
                    this.version++;
                }

                case OrderUpdatedEvent(var oid, var updates, var ts) -> {
                    this.status = "UPDATED";
                    this.version++;
                }

                case OrderCancelledEvent(var oid, var reason, var ts) -> {
                    this.status = "CANCELLED";
                    this.version++;
                }
            }

            history.add(event);
        }

        private void applyAndPersist(OrderEvent event) {
            apply(event);
            eventStore.append(orderId, event, version);
        }

        public String getState() {
            return "Order[%s] customer=%s items=%s amount=$%.2f status=%s v=%d".formatted(
                orderId, customerId, items, amount, status, version
            );
        }

        public long getVersion() {
            return version;
        }
    }

    // ── Usage Example ────────────────────────────────────────────────────────

    public static void main(String[] args) throws Exception {
        var eventStore = new EventStore<OrderEvent>();

        // Create aggregate
        var order = new OrderAggregate("order-123", eventStore);

        // Execute commands (generates events)
        System.out.println("=== Executing Commands ===\n");

        order.createOrder("customer-456", "item-1,item-2", 99.99);
        Thread.sleep(50);

        order.cancelOrder("Customer request");
        Thread.sleep(50);

        System.out.println("\nCurrent state: " + order.getState());

        // ── Event Replay ─────────────────────────────────────────────────────

        System.out.println("\n=== Event Replay ===\n");

        var replayedOrder = new OrderAggregate("order-123", eventStore);
        replayedOrder.loadFromHistory();

        System.out.println("Replayed state: " + replayedOrder.getState());
        System.out.println("Version: " + replayedOrder.getVersion());

        // ── Snapshot Strategy ────────────────────────────────────────────────

        System.out.println("\n=== Snapshot Strategy ===\n");

        // After N events, save snapshot
        var snapshotVersion = 1;
        var replayedFromSnapshot = new OrderAggregate("order-123", eventStore);

        // Load snapshot (simulated)
        replayedFromSnapshot.replayFromVersion(snapshotVersion);

        System.out.println("Replayed from snapshot v" + snapshotVersion + ": " + replayedFromSnapshot.getState());
    }
}
```

## Part 4: Event Replay and Projection

### Pattern: Rebuilding Read Models

```java
package io.github.seanchatmangpt.jotp.examples.events;

import io.github.seanchatmangpt.jotp.*;
import java.util.List;
import java.util.function.Consumer;

/**
 * EventReplay - Demonstrates rebuilding projections from event store
 * for disaster recovery and read model regeneration.
 */
public class EventReplay {

    public static class ProjectionManager {
        private final EventStore<OrderEvent> eventStore;
        private final List<Consumer<OrderEvent>> projections = new ArrayList<>();

        public ProjectionManager(EventStore<OrderEvent> eventStore) {
            this.eventStore = eventStore;
        }

        public void registerProjection(Consumer<OrderEvent> projection) {
            projections.add(projection);
        }

        public void rebuildAllProjections() {
            System.out.println("[PROJECTION] Starting rebuild...");

            var allEvents = eventStore.getAllEvents();

            for (var storedEvent : allEvents) {
                for (var projection : projections) {
                    try {
                        projection.accept(storedEvent.event());
                    } catch (Exception e) {
                        System.err.println("Projection failed: " + e.getMessage());
                    }
                }
            }

            System.out.println("[PROJECTION] Rebuild complete: " + allEvents.size() + " events");
        }

        public void rebuildFromSnapshot(long snapshotVersion) {
            System.out.println("[PROJECTION] Rebuilding from snapshot v" + snapshotVersion + "...");

            var events = eventStore.getAllEvents().stream()
                .filter(e -> e.version() >= snapshotVersion)
                .toList();

            for (var storedEvent : events) {
                for (var projection : projections) {
                    try {
                        projection.accept(storedEvent.event());
                    } catch (Exception e) {
                        System.err.println("Projection failed: " + e.getMessage());
                    }
                }
            }

            System.out.println("[PROJECTION] Rebuild complete: " + events.size() + " events");
        }
    }

    public static void main(String[] args) throws Exception {
        var eventStore = new EventStore<OrderEvent>();
        var projectionManager = new ProjectionManager(eventStore);

        // Register projections
        projectionManager.registerProjection(event -> {
            if (event instanceof OrderCreatedEvent) {
                System.out.println("[PROJECTION 1] Order created");
            }
        });

        projectionManager.registerProjection(event -> {
            if (event instanceof OrderCancelledEvent) {
                System.out.println("[PROJECTION 2] Order cancelled");
            }
        });

        // Store events
        eventStore.append("order-1", new OrderCreatedEvent("order-1", "cust-1", "item-1", 99.99, System.currentTimeMillis()), 1);
        eventStore.append("order-1", new OrderCancelledEvent("order-1", "reason", System.currentTimeMillis()), 2);

        // Rebuild projections
        projectionManager.rebuildAllProjections();
    }
}
```

## Common Pitfalls

### 1. Event Ordering

<Callout type="error">
**Don't rely on event order across aggregates**

```java
// BAD: Assuming global ordering
eventStore.getAllEvents().stream()
    .sorted(Comparator.comparing(StoredEvent::timestamp));

// GOOD: Order within aggregate only
eventStore.getStream(aggregateId).stream()
    .sorted(Comparator.comparing(StoredEvent::version));
```
</Callout>

### 2. Missing Event Handlers

<Callout type="error">
**Handle all event types in projections**

```java
// BAD: Missing case causes event to be ignored
switch (event) {
    case OrderCreated e -> handleCreate(e);
    // Missing OrderUpdated!
}

// GOOD: Exhaustive pattern matching
switch (event) {
    case OrderCreated e -> handleCreate(e);
    case OrderUpdated e -> handleUpdate(e);
    case OrderCancelled e -> handleCancel(e);
}
```
</Callout>

## Testing Event-Driven Systems

```java
@Test
void cqrsCommandUpdatesReadModel() throws Exception {
    var eventManager = new EventManager<OrderEvent>();
    var readModel = new ReadModelProjection();
    var commandHandler = new CommandHandler(eventManager);
    var queryHandler = new QueryHandler(readModel);

    eventManager.subscribeHandler(e -> readModel.handle(e));

    // Execute command
    commandHandler.handle(new CreateOrder("order-1", "cust-1", "item-1", 99.99));
    Thread.sleep(50);

    // Query read model
    var order = queryHandler.handle(new GetOrder("order-1"));

    assertThat(order.orderId()).isEqualTo("order-1");
    assertThat(order.status()).isEqualTo("ACTIVE");
}

@Test
void eventReplayRebuildsState() throws Exception {
    var eventStore = new EventStore<OrderEvent>();

    // Store events
    eventStore.append("order-1", new OrderCreatedEvent(...), 1);
    eventStore.append("order-1", new OrderCancelledEvent(...), 2);

    // Replay events
    var aggregate = new OrderAggregate("order-1", eventStore);
    aggregate.loadFromHistory();

    assertThat(aggregate.getStatus()).isEqualTo("CANCELLED");
    assertThat(aggregate.getVersion()).isEqualTo(2);
}
```
